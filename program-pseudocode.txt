    FUNCTION getComputerChoice
        COMMENT the function, getComputerChoice randomly returns computer choice of either "rock", "paper", or "scissors"
        SET randomNumber to a randomly generated number between 1 and 3
        INPUT choice
        IF randomNumber is 1 THEN
            choice is "Scissors"
        ELSE IF randomNumber is 2 THEN
            choice is "rock"
        ELSE IF randomNumber is 3 THEN
            choice is "paper"
        END IF
        OUTPUT choice
    END FUNCTION

            choice is "rock"
        ELSE IF playerInput is 2 THEN
            choice is "paper"

    END FUNCTION

            IF computerChoice is "rock" AND unless humanChoice is "rock" OR "paper" THEN
